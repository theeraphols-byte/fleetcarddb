(เช่น 15/06/2026) ไปเป็น Date Object เพื่อเปรียบเทียบหาค่ามากที่สุด const d = new Date(parseInt(parts[2], 10), parseInt(parts[1], 10) - 1, parseInt(parts[0], 10)); if (!isNaN(d.getTime())) { if (!maxDateObj || d > maxDateObj) { maxDateObj = d; } } } }); if (maxDateObj) { const monthsTh = ['ม.ค.', 'ก.พ.', 'มี.ค.', 'เม.ย.', 'พ.ค.', 'มิ.ย.', 'ก.ค.', 'ส.ค.', 'ก.ย.', 'ต.ค.', 'พ.ย.', 'ธ.ค.']; const day = maxDateObj.getDate(); const monthStr = monthsTh[maxDateObj.getMonth()]; const yearTh = maxDateObj.getFullYear() + 543; // เปลี่ยน ค.ศ. เป็น พ.ศ. ให้พนักงานอ่านง่าย // อัปเดตข้อความตามเงื่อนไข "ข้อมูล Post ล่าสุด ณ วันที่" displayEl.innerText = `ข้อมูล Post ล่าสุด ณ วันที่ ${day} ${monthStr} ${yearTh} | อัปเดตเรียลไทม์`; } else { displayEl.innerText = "ข้อมูล Post ล่าสุด ณ วันที่ — | โปรดตรวจสอบรูปแบบวันที่ใน Google Sheets (ต้องเป็น วว/ดด/ปปปป ค.ศ.)"; } } // ระบบค้นหาและเข้าสู่ระบบ Dashboard ด้วย Fleet Card function doLogin(){ const inputEl = document.getElementById('cardInput'); const errBox = document.getElementById('loginError'); if(!inputEl || !errBox) return; const input = String(inputEl.value).trim().replace(/[^0-9]/g, ''); if(!input) return; if(!QUOTA_DATA[input]){ errBox.innerText = '❌ ไม่พบหมายเลข Fleet Card นี้ในระบบ กรุณาตรวจสอบเลขอีกครั้ง'; errBox.style.display = 'block'; return; } errBox.style.display = 'none'; currentCard = input; currentTxn = TXN_DATA.filter(t => t.card === currentCard); currentPage = 1; document.getElementById('loginPage').style.display = 'none'; document.getElementById('dashPage').style.display = 'block'; renderDashboard(); } function doLogout(){ currentCard = ''; currentTxn = []; const inputEl = document.getElementById('cardInput'); if(inputEl) inputEl.value = ''; document.getElementById('dashPage').style.display = 'none'; document.getElementById('loginPage').style.display = 'flex'; } function renderDashboard(){ const info = QUOTA_DATA[currentCard] || {}; const firstChar = info.name ? info.name.trim().charAt(0) : '—'; document.getElementById('empAvatar').innerText = firstChar; document.getElementById('empName').innerText = info.name || 'ไม่ทราบชื่อ'; document.getElementById('empSub').innerText = `${info.company || 'TD'} · ${info.team || 'ไม่ระบุ'} · ${info.position || 'พนักงาน'}`; document.getElementById('empBadge').innerText = `ทะเบียนรถ: ${info.plate || '—'}`; const q3Txns = currentTxn.filter(t => t.year === 2026 && (t.month === 7 || t.month === 8 || t.month === 9)); const usedQty = q3Txns.reduce((sum, t) => sum + t.qty, 0); const totalQuota = info.q3Quota || 0; const remainQty = totalQuota - usedQty; document.getElementById('kpiQuota').innerText = totalQuota.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2}); document.getElementById('kpiUsed').innerText = usedQty.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2}); document.getElementById('kpiRemain').innerText = remainQty.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2}); const remainCard = document.getElementById('kpiRemainCard'); if(remainQty < 0) { remainCard.style.background = '#FEF2F2'; remainCard.querySelector('.kpi-lbl').innerText = 'เติมน้ำมันเกินโควต้า (ลิตร)'; remainCard.querySelector('.kpi-lbl').style.color = '#B91C1C'; } else { remainCard.style.background = '#fff'; remainCard.querySelector('.kpi-lbl').innerText = 'คงเหลือ'; remainCard.querySelector('.kpi-lbl').style.color = '#888'; } let pct = totalQuota > 0 ? (usedQty / totalQuota) * 100 : 0; if(pct > 100) pct = 100; document.getElementById('kpiBar').style.width = `${pct}%`; document.getElementById('kpiUsedPct').innerText = `ใช้ไปแล้ว ${pct.toFixed(1)}% ของโควต้ารวม`; renderMonthlySummary(); renderTxnPage(); } function renderMonthlySummary(){ const info = QUOTA_DATA[currentCard] || {}; const tbody = document.getElementById('monthBody'); if(!tbody) return; const targetMonths = [ { label: 'เมษายน 2569', m: 4, y: 2026, qQuota: info.aprQty || 0 }, { label: 'พฤษภาคม 2569', m: 5, y: 2026, qQuota: info.mayQty || 0 }, { label: 'มิถุนายน 2569', m: 6, y: 2026, qQuota: info.junQty || 0 } ]; let html = ''; let hasData = false; targetMonths.forEach(item => { const mTxns = currentTxn.filter(t => t.month === item.m && t.year === item.y); const sumQty = mTxns.reduce((sum, t) => sum + t.qty, 0); const sumExcl = mTxns.reduce((sum, t) => sum + t.excl, 0); const avgRate = sumQty > 0 ? avgRate.toFixed(2) : '0.00'} ${sumExcl.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2})} `; }); tbody.innerHTML = hasData ? html : `ไม่มีประวัติการเติมน้ำมันในไตรมาสนี้`; } function renderTxnPage(){ const tbody = document.getElementById('txnBody'); if(!tbody) return; if(!currentTxn.length){ tbody.innerHTML = `ไม่พบประวัติการเติมน้ำมันของบัตรนี้ในระบบ`; document.getElementById('paginationWrap').innerHTML = ''; return; } // เรียงลำดับจากวันที่ล่าสุดขึ้นก่อน (Latest Post First) let currentTxnsSorted = [...currentTxn].sort((a,b) => { const pA = a.dateStr.split('/'); const pB = b.dateStr.split('/'); const dA = new Date(pA[2], pA[1]-1, pA[0]); const dB = new Date(pB[2], pB[1]-1, pB[0]); return dB - dA; }); const totalRows = currentTxnsSorted.length; const totalPages = Math.ceil(totalRows / PAGE_SIZE); if(currentPage > totalPages) currentPage = totalPages; const startIdx = (currentPage - 1) * PAGE_SIZE; const endIdx = Math.min(startIdx + PAGE_SIZE, totalRows); const pageData = currentTxnsSorted.slice(startIdx, endIdx); let html = ''; pageData.forEach(t => { const rate = t.qty > 0 ?
